496	                                .font(PanelTypography.body(settings))
497	                                .foregroundStyle(tokens.textPrimary)

499	                        } else {

501	                                .markdownTheme(.clippy(tokens: tokens, settings: settings))
502	                        }
503	                    } else {
504	                        Text(message.text.isEmpty ? " " : message.text)
505	                            .font(PanelTypography.body(settings))
506	                            .foregroundStyle(tokens.isDark ? Color.white : Color(nsColor: .labelColor).opacity(0.9))
